제네릭 메서드

  • 제네릭 메소드란 메소드의 선언부에 타입 변수를 사용한 메소드를 의미하며 타입 변수의 선언은 메소드 선언부에서 반환 타입 바로 앞에 위치함

제네릭 메서드 선언 클래스

xxxxxxxxxx
// 제네릭 메서드는 메서드에만 적용되는 제네릭 타입이므로, 클래스에 제네릭을 안써도 무방함 (물론, 제네릭 클래스여도 상관없음)
class GenericMethodContainer {
    // 메서드에 파라미터 타입을 제네릭 타입으로 지정 가능
    public <T> void print(T value) {
        System.out.println(value);
    }
    // static 키워드와 상관없이 반환 타입 옆에만 다이아몬드랑 타입 이름 써주면 됨
    public static <T> void printArrayStatic(T[] array) {
        for (T item : array) {
            System.out.print(item + " ");
        }
        System.out.println();
    }
}

활용 코드

xxxxxxxxxx
GenericMethodContainer container = new GenericMethodContainer();
// 이렇게 써줘도 되지만
container.<String>print("Hello");
// 타입 추론되므로 그냥 값만 써줘도 상관 없음
container.print("Hello");
// 마찬가지
GenericMethodContainer.<Integer>printArrayStatic(new Integer[]{ 1, 2, 3 });
GenericMethodContainer.printArrayStatic(new String[]{ "Hello", "World" });

 

Bounded Type Parameter

  • 제네릭 타입으로 설정할 수 있는 타입 종류를 제한하고 싶은 경우 사용

    • 특정 클래스를 상속받거나 인터페이스를 구현하고 있는 타입으로 제네릭 타입을 제한할 수 있음
xxxxxxxxxx
// 제네릭 타입을 지정하되, Number 클래스를 상속받는 타입으로 제한
class NumberBox<T extends Number> {
    private T numberValue;
    public void set(T numberValue) { 
        this.numberValue = numberValue;
    }
    public T get() {
        return numberValue; 
    }
}

활용 코드

xxxxxxxxxx
// Integer, Double은 Number 추상 클래스를 상속받으므로 타입 지정 가능

NumberBox<Double> box2 = new NumberBox<>();
// 타입 지정 불가!
// NumberBox<String> box3 = new NumberBox<>();
  • 특정 인터페이스를 구현하는 타입으로 제한하고 싶다고 하더라도 T extends 타입으로 씀 (implements 안 씀)
xxxxxxxxxx
interface I {}
// "T implements I"가 아님을 유의
class GenericDemo<T extends I> {
    T impl;
}
class Impl implements I {}

활용 코드

xxxxxxxxxx

  • 여러 클래스, 인터페이스를 구현하는 타입으로 제네릭 타입을 제한하는 것도 가능함
xxxxxxxxxx
class A {}
interface I1 {}
interface I2 {}
class B extends A implements I1, I2 {}
class C extends A implements I1 {}
// A 클래스를 상속받고 I1 인터페이스를 구현하고 있는 타입으로 제네릭 타입을 제한
class GenericDemo1<T extends A & I1> {}
// A 클래스를 상속받고 I1, I2 인터페이스를 구현하고 있는 타입으로 제네릭 타입을 제한
class GenericDemo2<T extends A & I1 & I2> {}

타입 지우기 (type erasure)

  • 기본적으로 제네릭 타입 정보는 컴파일 타임까지만 유지되며 프로그램이 실행되면(=런타임) 제네릭 타입 정보가 사라짐(=지워짐, 타입 지우기, type erasure)
xxxxxxxxxx
class DemoClass<T> {
    T value;
    public DemoClass(T value) {
        this.value = value;
    }
}
  • 위의 클래스에 제네릭 타입을 적용하여 다음과 같이 코드를 작성해도
xxxxxxxxxx
DemoClass<Integer> dc1 = new DemoClass<>();
DemoClass<String> dc2 = new DemoClass<>();
  • 실제 실행되는 시점에는 타입 정보(T)가 Object로 바뀌어서 실행됨
xxxxxxxxxx
// 프로그램 실행과 동시에 내부적으로는 다음과 같이 완전히 똑같은 형태의 객체가 생성됨을 유의
DemoClass<Object> dc1 = new DemoClass<>();
DemoClass<Object> dc2 = new DemoClass<>();
  • Bounded Type Parameter를 사용한 경우에는 상속 대상 클래스 혹은 인터페이스로 타입이 변경되어 실행됨

  • 타입 지우기에 따라 타입 매개변수가 나타내는 타입의 객체 혹은 배열을 생성이 불가능함
xxxxxxxxxx
class GenericTypeErasureDemo<T> {
    T value;
    
    public GenericTypeErasureDemo(T value) {
        this.value = value;
        // T 타입에 대한 정보는 컴파일 시점에만 보전되고 해당 객체 생성이 이루어지는 런타임에는 타입 정보가 없으므로,
        // T 타입의 객체 생성 불가 (컴파일러는 T라는 클래스 생성자를 호출하는 것으로 인식)
        this.value = new T();
        T typeValue = new T();
        // T 타입의 객체 배열 생성도 불가
        T[] typeArray = new T[10];
    }
}
  • 제네릭 타입과 관련된 객체 및 객체 배열과 관련된 workaround가 존재함
xxxxxxxxxx
class GenericTypeErasureWorkaround<T1, T2> {
    T1 value;
    T2[] arr;


    // 배열 타입 캐스팅 관련 경고 없애기, 꼭 안 써줘도 됨
    @SuppressWarnings("unchecked")
    public GenericTypeErasureWorkaround(Class<T1> clazz, Class<T2> arrayClazz, int arraySize) throws InstantiationException, IllegalAccessException {
        // 제네릭 타입의 객체 생성
        // 해당 클래스에 기본 생성자가 있어야만 호출 가능하는 단점 있음
        // https://stackoverflow.com/questions/11822621/instantiationexception-for-newinstance
        value = clazz.newInstance();
        
        // 제네릭 타입의 객체 배열 생성
        // https://stackoverflow.com/questions/529085/how-to-create-a-generic-array-in-java
        // https://stackoverflow.com/questions/39765448/classcast-exception-in-java-cannot-cast-to-ljava-lang-integer
        arr = (T2[]) Array.newInstance(arrayClazz, arraySize);
    }


    public T1 getValue() {
        return this.value;
    }


    public void setArrayItem(int index, T2 value) {
        arr[index] = value;
    }


    public T2 getArrayItem(int index) {
        return arr[index];
    }


    public T2[] getArray() {
        return arr;
    }
}


class Person {}

class GrandGrandFather {}

class Father extends GrandFather {}

class MyBox<T> {
    private T value;
    public T getValue() {
        return this.value;
    }
    public void setValue(T value) {
        this.value = value;
    }
}


public class CovarianceDemo {
    public static void main(String args[]) {
        String s = "Hello";
        // 업캐스팅
        Object o = s;


        String[] sArray = { "Hello" };
        // 적법한 대입으로 간주하여 허용
        Object[] oArray = sArray;


        Integer[] iArray = { 1, 2, 3 };
        // Number 추상클래스는 다양한 숫자 타입의 Wrapper 클래스(Integer, Long, Double, Float 등등)가 상속받음
        Number[] nArray = iArray;
        // 문제 없음
        nArray[0] = 100;
        // 컴파일 시점에서는 문제 없음, 그러나 실행시 Heap 메모리 오염으로 인한 ArrayStoreException이 발생함
        // nArray[1] = 1.0;


        // 제네릭 타입의 경우 위와 같은 시나리오 발생이 불가능함
        ArrayList<Integer> integerArrayList = new ArrayList<Integer>(Arrays.asList(1, 2, 3));
        // 밑의 대입을 허용하지 않으므로, 제네릭 타입의 업캐스팅을 기대할 수 없음
        // ArrayList<Number> numberArrayList = integerArrayList;
        // (*) Number => Intger의 관계는 성립하지만, ArrayList<Number> => ArrayList<Integer>의 관계는 성립하지 않음 (이 둘은 어떠한 관계도 없음)
        // 완전히 같은 제네릭 타입을 가진 변수에만 대입 가능
        ArrayList<Integer> anotherIntegerArrayList = integerArrayList;


        // 만약 제네릭 타입 업캐스팅을 기대하는 경우 와일드카드를 사용
        method(integerArrayList);
        method(new ArrayList<Double>(Arrays.asList(1.1, 2.2, 3.3)));
        method(new ArrayList<Float>(Arrays.asList(1.1F, 2.2F, 3.3F)));


        // 단순히 콜렉션 문제가 아니라 직접 정의한 제네릭 클래스에서도 위의 성질은 적용됨
        MyBox<GrandGrandFather> grandGrandFatherMyBox;
        grandGrandFatherMyBox = new MyBox<GrandGrandFather>();
        // 모두 대입 불가
        // grandGrandFatherMyBox = new MyBox<GrandFather>();
        // grandGrandFatherMyBox = new MyBox<Father>();


        MyBox<GrandChild> grandChildMyBox = new MyBox<GrandChild>();
        grandChildMyBox.setValue(new GrandChild());
        // 와일드카드를 사용하여 제네릭 타입 설정
        MyBox<? extends GrandGrandFather> grandGrandFatherMyBoxWithWildcard;
        // 와일드카드를 사용하였으므로 다음과 같은 대입을 허용하지만
        grandGrandFatherMyBoxWithWildcard = grandChildMyBox;
        // 값의 쓰기는 허용하지 않고
        // grandGrandFatherMyBoxWithWildcard.setValue(new GrandChild());

    public static void method(ArrayList<? extends Number> arrayList) {

        for(Number n : arrayList) {

            System.out.println(n.doubleValue());
        }


        // Q) 다음과 같이 값 쓰기 연산이 불가능한(=허용하지 않는) 이유?
        // 힌트) 앞서 배열 전체의 업캐스팅 이후 발생한 문제점을 생각해보기
        // arrayList.add(1);


        // A) 전달받은 arrayList가 ArrayList<Double> 혹은 ArrayList<Float>과 같은 다른 Number 상속 클래스일 수도 있으므로
        // Heap 오염 문제를 막기 위해서 쓰기 작업을 원천 차단
    }
}
